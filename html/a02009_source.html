88 static void
89 fade(unsigned char l)
90 {
91  volatile int i;
92  int k, j;
93  for(k = 0; k < 800; ++k) {
94  j = k > 400 ? 800 - k : k;
95 
96  leds_on(l);
97  for(i = 0; i < j; ++i) {
98  asm("nop");
99  }
100  leds_off(l);
101  for(i = 0; i < 400 - j; ++i) {
102  asm("nop");
103  }
104  }
105 }

107 static void
108 set_rf_params(void)
109 {
110  uint16_t short_addr;
111  uint8_t ext_addr[8];
112 
113  ieee_addr_cpy_to(ext_addr, 8);
114 
115  short_addr = ext_addr[7];
116  short_addr |= ext_addr[6] << 8;
117 
118  /* Populate linkaddr_node_addr. Maintain endianness */
119  memcpy(&linkaddr_node_addr, &ext_addr[8 - LINKADDR_SIZE], LINKADDR_SIZE);
120 
121 #if STARTUP_CONF_VERBOSE
122  {
123  int i;
124  printf("Rime configured with address ");
125  for(i = 0; i < LINKADDR_SIZE - 1; i++) {
126  printf("%02x:", linkaddr_node_addr.u8[i]);
127  }
128  printf("%02x\n", linkaddr_node_addr.u8[i]);
129  }
130 #endif
131 
132  NETSTACK_RADIO.set_value(RADIO_PARAM_PAN_ID, IEEE802154_PANID);
133  NETSTACK_RADIO.set_value(RADIO_PARAM_16BIT_ADDR, short_addr);
134  NETSTACK_RADIO.set_value(RADIO_PARAM_CHANNEL, CC2538_RF_CHANNEL);
135  NETSTACK_RADIO.set_object(RADIO_PARAM_64BIT_ADDR, ext_addr, 8);
136 }

138 /**
139  * \brief Main routine for the cc2538dk platform
140  */
141 int
142 main(void)
143 {
144  nvic_init();
145  ioc_init();
146  sys_ctrl_init();
147  clock_init();
148  lpm_init();
149  rtimer_init();
150  gpio_init();
151 
152  leds_init();
153  fade(LEDS_YELLOW);
154 
155  process_init();
156 
157  watchdog_init();
158  button_sensor_init();
159 
160  /*
161  * Character I/O Initialisation.
162  * When the UART receives a character it will call serial_line_input_byte to
163  * notify the core. The same applies for the USB driver.
164  *
165  * If slip-arch is also linked in afterwards (e.g. if we are a border router)
166  * it will overwrite one of the two peripheral input callbacks. Characters
167  * received over the relevant peripheral will be handled by
168  * slip_input_byte instead
169  */
170 #if UART_CONF_ENABLE
171  uart_init(0);
172  uart_init(1);
173  uart_set_input(SERIAL_LINE_CONF_UART, serial_line_input_byte);
174 #endif
175 
176 #if USB_SERIAL_CONF_ENABLE
177  usb_serial_init();
178  usb_serial_set_input(serial_line_input_byte);
179 #endif
180 
181  serial_line_init();
182 
183  INTERRUPTS_ENABLE();
184  fade(LEDS_GREEN);
185 
186  PUTS(CONTIKI_VERSION_STRING);
187  PUTS(BOARD_STRING);
188 #if STARTUP_CONF_VERBOSE
189  soc_print_info();
190 #endif
191 
192  PRINTF(" Net: ");
193  PRINTF("%s\n", NETSTACK_NETWORK.name);
194  PRINTF(" MAC: ");
195  PRINTF("%s\n", NETSTACK_MAC.name);
196  PRINTF(" RDC: ");
197  PRINTF("%s\n", NETSTACK_RDC.name);
198 
199  /* Initialise the H/W RNG engine. */
200  random_init(0);
201 
202  udma_init();
203 
204  process_start(&etimer_process, NULL);
205  ctimer_init();
206 
207  set_rf_params();
208 
209 #if CRYPTO_CONF_INIT
210  crypto_init();
211  crypto_disable();
212 #endif
213 
214  netstack_init();
215 
216 #if NETSTACK_CONF_WITH_IPV6
217  memcpy(&uip_lladdr.addr, &linkaddr_node_addr, sizeof(uip_lladdr.addr));
218  queuebuf_init();
219  process_start(&tcpip_process, NULL);
220 #endif /* NETSTACK_CONF_WITH_IPV6 */
221 
222  adc_init();
223 
224  process_start(&sensors_process, NULL);
225 
226  energest_init();
227  ENERGEST_ON(ENERGEST_TYPE_CPU);
228 
229  autostart_start(autostart_processes);
230 
231  watchdog_start();
232  fade(LEDS_ORANGE);
233 
234  while(1) {
235  uint8_t r;
236  do {
237  /* Reset watchdog and handle polls and events */
238  watchdog_periodic();
239 
240  r = process_run();
241  } while(r > 0);
242 
243  /* We have serviced all pending events. Enter a Low-Power mode. */
244  lpm_enter();
245  }
246 }
