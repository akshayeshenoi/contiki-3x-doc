80 static void
81 fade(unsigned char l)
82 {
83  volatile int i;
84  int k, j;
85  for(k = 0; k < 800; ++k) {
86  j = k > 400 ? 800 - k : k;
87 
88  leds_on(l);
89  for(i = 0; i < j; ++i) {
90  __asm("nop");
91  }
92  leds_off(l);
93  for(i = 0; i < 400 - j; ++i) {
94  __asm("nop");
95  }
96  }
97 }

99 static void
100 set_rf_params(void)
101 {
102  uint16_t short_addr;
103  uint8_t ext_addr[8];
104  radio_value_t val = 0;
105 
106  ieee_addr_cpy_to(ext_addr, 8);
107 
108  short_addr = ext_addr[7];
109  short_addr |= ext_addr[6] << 8;
110 
111  /* Populate linkaddr_node_addr. Maintain endianness */
112  memcpy(&linkaddr_node_addr, &ext_addr[8 - LINKADDR_SIZE], LINKADDR_SIZE);
113 
114  NETSTACK_RADIO.set_value(RADIO_PARAM_PAN_ID, IEEE802154_PANID);
115  NETSTACK_RADIO.set_value(RADIO_PARAM_16BIT_ADDR, short_addr);
116  NETSTACK_RADIO.set_value(RADIO_PARAM_CHANNEL, RF_CORE_CHANNEL);
117  NETSTACK_RADIO.set_object(RADIO_PARAM_64BIT_ADDR, ext_addr, 8);
118 
119  NETSTACK_RADIO.get_value(RADIO_PARAM_CHANNEL, &val);
120  printf(" RF: Channel %d\n", val);
121 
122 #if STARTUP_CONF_VERBOSE
123  {
124  int i;
125  printf(" Link layer addr: ");
126  for(i = 0; i < LINKADDR_SIZE - 1; i++) {
127  printf("%02x:", linkaddr_node_addr.u8[i]);
128  }
129  printf("%02x\n", linkaddr_node_addr.u8[i]);
130  }
131 #endif
132 
133  /* also set the global node id */
134  node_id = short_addr;
135  printf(" Node ID: %d\n", node_id);
136 }

138 /**
139  * \brief Main function for CC26xx-based platforms
140  *
141  * The same main() is used for all supported boards
142  */
143 int
144 main(void)
145 {
146  /* Enable flash cache and prefetch. */
147  ti_lib_vims_mode_set(VIMS_BASE, VIMS_MODE_ENABLED);
148  ti_lib_vims_configure(VIMS_BASE, true, true);
149 
150  ti_lib_int_master_disable();
151 
152  /* Set the LF XOSC as the LF system clock source */
153  oscillators_select_lf_xosc();
154 
155  lpm_init();
156 
157  board_init();
158 
159  gpio_interrupt_init();
160 
161  leds_init();
162 
163  /*
164  * Disable I/O pad sleep mode and open I/O latches in the AON IOC interface
165  * This is only relevant when returning from shutdown (which is what froze
166  * latches in the first place. Before doing these things though, we should
167  * allow software to first regain control of pins
168  */
169  ti_lib_pwr_ctrl_io_freeze_disable();
170 
171  fade(LEDS_RED);
172 
173  ti_lib_int_master_enable();
174 
175  soc_rtc_init();
176  clock_init();
177  rtimer_init();
178 
179  watchdog_init();
180  process_init();
181 
182  random_init(0x1234);
183 
184  /* Character I/O Initialisation */
185 #if CC26XX_UART_CONF_ENABLE
186  cc26xx_uart_init();
187 #endif
188 
189  serial_line_init();
190 
191  printf("Starting " CONTIKI_VERSION_STRING "\n");
192  printf("With DriverLib v%u.%u\n", DRIVERLIB_RELEASE_GROUP,
193  DRIVERLIB_RELEASE_BUILD);
194  printf(BOARD_STRING "\n");
195  printf("IEEE 802.15.4: %s, Sub-GHz: %s, BLE: %s, Prop: %s\n",
196  ti_lib_chipinfo_supports_ieee_802_15_4() == true ? "Yes" : "No",
197  ti_lib_chipinfo_chip_family_is_cc13xx() == true ? "Yes" : "No",
198  ti_lib_chipinfo_supports_ble() == true ? "Yes" : "No",
199  ti_lib_chipinfo_supports_proprietary() == true ? "Yes" : "No");
200 
201  process_start(&etimer_process, NULL);
202  ctimer_init();
203 
204  energest_init();
205  ENERGEST_ON(ENERGEST_TYPE_CPU);
206 
207  fade(LEDS_YELLOW);
208 
209  printf(" Net: ");
210  printf("%s\n", NETSTACK_NETWORK.name);
211  printf(" MAC: ");
212  printf("%s\n", NETSTACK_MAC.name);
213  printf(" RDC: ");
214  printf("%s", NETSTACK_RDC.name);
215 
216  if(NETSTACK_RDC.channel_check_interval() != 0) {
217  printf(", Channel Check Interval: %u ticks",
218  NETSTACK_RDC.channel_check_interval());
219  }
220  printf("\n");
221 
222  netstack_init();
223 
224  set_rf_params();
225 
226 #if NETSTACK_CONF_WITH_IPV6
227  memcpy(&uip_lladdr.addr, &linkaddr_node_addr, sizeof(uip_lladdr.addr));
228  queuebuf_init();
229  process_start(&tcpip_process, NULL);
230 #endif /* NETSTACK_CONF_WITH_IPV6 */
231 
232  fade(LEDS_GREEN);
233 
234  process_start(&sensors_process, NULL);
235 
236  autostart_start(autostart_processes);
237 
238  watchdog_start();
239 
240  fade(LEDS_ORANGE);
241 
242  while(1) {
243  uint8_t r;
244  do {
245  r = process_run();
246  watchdog_periodic();
247  } while(r > 0);
248 
249  /* Drop to some low power mode */
250  lpm_drop();
251  }
252 }
