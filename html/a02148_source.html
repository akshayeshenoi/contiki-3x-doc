49 /*---------------------------------------------------------------------------*/
50 static aux_consumer_module_t als_aux = {
51  .clocks = AUX_WUC_ADI_CLOCK | AUX_WUC_ANAIF_CLOCK | AUX_WUC_SMPH_CLOCK
52 };
53 /*---------------------------------------------------------------------------*/
54 static int
55 config(int type, int enable)
56 {
57  switch(type) {
58  case SENSORS_HW_INIT:
59  ti_lib_ioc_pin_type_gpio_output(BOARD_IOID_ALS_PWR);
60  break;
61  case SENSORS_ACTIVE:
62  ti_lib_rom_ioc_pin_type_gpio_output(BOARD_IOID_ALS_PWR);
63  ti_lib_rom_ioc_port_configure_set(BOARD_IOID_ALS_OUT, IOC_PORT_GPIO,
64  IOC_STD_OUTPUT);
65  ti_lib_rom_ioc_pin_type_gpio_input(BOARD_IOID_ALS_OUT);
66 
67  if(enable) {
68  ti_lib_gpio_set_dio(BOARD_IOID_ALS_PWR);
69  aux_ctrl_register_consumer(&als_aux);
70  ti_lib_aux_adc_select_input(ADC_COMPB_IN_AUXIO7);
71  clock_delay_usec(2000);
72  } else {
73  ti_lib_gpio_clear_dio(BOARD_IOID_ALS_PWR);
74  aux_ctrl_unregister_consumer(&als_aux);
75  }
76  break;
77  default:
78  break;
79  }
80  return 1;
81 }
82 /*---------------------------------------------------------------------------*/
83 static int
84 value(int type)
85 {
86  int val;
87 
88  ti_lib_aux_adc_enable_sync(AUXADC_REF_VDDS_REL, AUXADC_SAMPLE_TIME_2P7_US,
89  AUXADC_TRIGGER_MANUAL);
90  ti_lib_aux_adc_gen_manual_trigger();
91  val = ti_lib_aux_adc_read_fifo();
92  ti_lib_aux_adc_disable();
93 
94  return val;
95 }
96 /*---------------------------------------------------------------------------*/

98 status(int type)

100  return 1;
101 }
