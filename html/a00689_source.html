44 void
45 uip_debug_ipaddr_print(const uip_ipaddr_t *addr)
46 {
47 #if NETSTACK_CONF_WITH_IPV6
48  uint16_t a;
49  unsigned int i;
50  int f;
51 #endif /* NETSTACK_CONF_WITH_IPV6 */
52  if(addr == NULL) {
53  PRINTA("(NULL IP addr)");
54  return;
55  }
56 #if NETSTACK_CONF_WITH_IPV6
57  if(ip64_addr_is_ipv4_mapped_addr(addr)) {
58  /*
59  * Printing IPv4-mapped addresses is done according to RFC 4291 [1]
60  *
61  * "An alternative form that is sometimes more
62  * convenient when dealing with a mixed environment
63  * of IPv4 and IPv6 nodes is x:x:x:x:x:x:d.d.d.d,
64  * where the 'x's are the hexadecimal values of the
65  * six high-order 16-bit pieces of the address, and
66  * the 'd's are the decimal values of the four
67  * low-order 8-bit pieces of the address (standard
68  * IPv4 representation)."
69  *
70  * [1] https://tools.ietf.org/html/rfc4291#page-4
71  */
72  PRINTA("::FFFF:%u.%u.%u.%u", addr->u8[12], addr->u8[13], addr->u8[14], addr->u8[15]);
73  } else {
74  for(i = 0, f = 0; i < sizeof(uip_ipaddr_t); i += 2) {
75  a = (addr->u8[i] << 8) + addr->u8[i + 1];
76  if(a == 0 && f >= 0) {
77  if(f++ == 0) {
78  PRINTA("::");
79  }
80  } else {
81  if(f > 0) {
82  f = -1;
83  } else if(i > 0) {
84  PRINTA(":");
85  }
86  PRINTA("%x", a);
87  }
88  }
89  }
90 #else /* NETSTACK_CONF_WITH_IPV6 */
91  PRINTA("%u.%u.%u.%u", addr->u8[0], addr->u8[1], addr->u8[2], addr->u8[3]);
92 #endif /* NETSTACK_CONF_WITH_IPV6 */
93 }
