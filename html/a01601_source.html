43 int
44 puts(const char *str)
45 {
46  int i;
47  if(str == NULL) {
48  return 0;
49  }
50  for(i = 0; i < strlen(str); i++) {
51  cc26xx_uart_write_byte(str[i]);
52  }
53  cc26xx_uart_write_byte('\n');
54 
55  /*
56  * Wait for the line to go out. This is to prevent garbage when used between
57  * UART on/off cycles
58  */
59  while(cc26xx_uart_busy() == UART_BUSY);
60 
61  return i;
62 }

64 unsigned int
65 dbg_send_bytes(const unsigned char *s, unsigned int len)
66 {
67  unsigned int i = 0;
68 
69  while(s && *s != 0) {
70  if(i >= len) {
71  break;
72  }
73  cc26xx_uart_write_byte(*s++);
74  i++;
75  }
76 
77  /*
78  * Wait for the buffer to go out. This is to prevent garbage when used
79  * between UART on/off cycles
80  */
81  while(cc26xx_uart_busy() == UART_BUSY);
82 
83  return i;
84 }
