52 eth_init(void)
53 {
54 #if !NETSTACK_CONF_WITH_IPV6
55  uip_ipaddr_t ip_addr;
56 
57 #define SET_IP_ADDR(x) \
58  uip_ipaddr(&ip_addr, x)
59 
60  SET_IP_ADDR(HOST_IP);
61  uip_sethostaddr(&ip_addr);
62 
63  SET_IP_ADDR(NETMASK_IP);
64  uip_setnetmask(&ip_addr);
65 
66  SET_IP_ADDR(GATEWAY_IP);
67  uip_setdraddr(&ip_addr);
68 
69 #if WITH_DNS
70  SET_IP_ADDR(NAMESERVER_IP);
71  uip_nameserver_update(&ip_addr, UIP_NAMESERVER_INFINITE_LIFETIME);
72 #endif
73 #endif
74 
75  process_start(&eth_process, NULL);
76 #if !NETSTACK_CONF_WITH_IPV6
77  /* Comment out the following line to disable DHCP and simply use the static
78  * IP configuration setup above.

82 }

85 PROCESS_THREAD(dhcp_process, ev, data)
86 {
87  PROCESS_BEGIN();
88 
89  dhcpc_init(uip_lladdr.addr, sizeof(uip_lladdr.addr));
90 
91  printf("Requesting DHCP configuration...\n");
92  dhcpc_request();
93 
94  while(1) {
95  PROCESS_WAIT_EVENT();
96 
97  if(ev == tcpip_event || ev == PROCESS_EVENT_TIMER) {
98  dhcpc_appcall(ev, data);
99  } else if(ev == PROCESS_EVENT_EXIT) {
100  process_exit(&dhcp_process);
101  }
102  }
103 
104  PROCESS_END();
105 }

107 void
108 dhcpc_configured(const struct dhcpc_state *s)
109 {
110  uip_sethostaddr(&s->ipaddr);
111  uip_setnetmask(&s->netmask);
112  uip_setdraddr(&s->default_router);
113  uip_nameserver_update(&s->dnsaddr, UIP_NAMESERVER_INFINITE_LIFETIME);
114  printf("DHCP configured:\n");
115  printf(" - Host IP: %d.%d.%d.%d\n", uip_ipaddr_to_quad(&s->ipaddr));
116  printf(" - Netmask: %d.%d.%d.%d\n", uip_ipaddr_to_quad(&s->netmask));
117  printf(" - Default router: %d.%d.%d.%d\n",
118  uip_ipaddr_to_quad(&s->default_router));
119  printf(" - DNS server: %d.%d.%d.%d\n", uip_ipaddr_to_quad(&s->dnsaddr));
120 }
