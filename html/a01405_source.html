38 /* RF230 register access is through SPI which transfers 8 bits address/8 bits data.
39  * ATmega128rfa1 registers are defined in I/O space, e.g. in gcc /include/avr/iom128rfa1.h
40  * A typical definition is #define TRX_STATUS _SFR_MEM8(0x141)
41  * Registers can be read with a macro, but the args for subregisters don't expand properly so the actual address
42  * is used with explicit _SFR_MEM8 in the subregister read/write routines.
43  */
