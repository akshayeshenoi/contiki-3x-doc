49 void
50 st_lib_sys_tick_handler(void)
51 {
52  ticks++;
53  if((ticks % CLOCK_SECOND) == 0) {
54  seconds++;
55  energest_flush();
56  }
57  HAL_IncTick();
58 
59  if(etimer_pending()) {
60  etimer_request_poll();
61  }
62 }
