78 struct channel *
79 channel_lookup(uint16_t channelno)
80 {
81  struct channel *c;
82  for(c = list_head(channel_list); c != NULL; c = list_item_next(c)) {
83  if(c->channelno == channelno) {
84  return c;
85  }
86  }
87  return NULL;
88 }
